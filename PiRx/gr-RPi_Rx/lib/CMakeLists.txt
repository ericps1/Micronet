########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})
list(APPEND RPi_Rx_sources
    RPi_Rx_impl.cc
    FSK_Rx_impl.cc
    )

add_library(gnuradio-RPi_Rx SHARED ${RPi_Rx_sources})
target_link_libraries(gnuradio-RPi_Rx ${Boost_LIBRARIES} ${GNURADIO_RUNTIME_LIBRARIES})
set_target_properties(gnuradio-RPi_Rx PROPERTIES DEFINE_SYMBOL "gnuradio_RPi_Rx_EXPORTS")

########################################################################
# Install built library files
########################################################################
install(TARGETS gnuradio-RPi_Rx
    LIBRARY DESTINATION lib${LIB_SUFFIX} # .so/.dylib file
    ARCHIVE DESTINATION lib${LIB_SUFFIX} # .lib file
    RUNTIME DESTINATION bin              # .dll file
)

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_RPi_Rx_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_RPi_Rx.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_RPi_Rx.cc
)

add_executable(test-RPi_Rx ${test_RPi_Rx_sources})

target_link_libraries(
  test-RPi_Rx
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-RPi_Rx
)

GR_ADD_TEST(test_RPi_Rx test-RPi_Rx)
